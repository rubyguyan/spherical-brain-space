```python
class NeuronDecision:
    """神经元决策系统"""
    
    def __init__(self):
        self.state_machine = NeuronStateMachine()  # 状态机（核心）
        self.random_factor = 0.1                   # 随机扰动（创新性）
        self.learning_weights = {}                 # 简单学习权重（适应性）
        self.memory = ShortTermMemory(capacity=10) # 短期记忆
    
    def decide_synapse_growth(self, context):
        """决定是否生长新突触"""
        # 基础得分（状态机）
        base_score = self.state_machine.evaluate(context)
        
        # 记忆调整
        memory_adjustment = self.memory.get_pattern_score('synapse_success')
        
        # 随机扰动
        random_adjustment = random.uniform(-self.random_factor, self.random_factor)
        
        # 最终决策
        final_score = base_score + memory_adjustment + random_adjustment
        return final_score > THRESHOLD
    
    def decide_movement(self, context):
        """决定移动方向"""
        # 饥饿驱动
        if context.satiety < 0.26:
            return self.move_towards_energy(context)
        
        # 连接驱动
        if context.connection_count == 0:
            return self.move_towards_neurons(context)
        
        # 探索驱动
        return self.random_walk(context)
    
    def decide_output(self, context):
        """决定输出内容"""
        # 基于当前状态和输入
        state_signal = self.state_machine.get_output_signal()
        input_response = self.process_recent_inputs(context)
        
        return self.combine_signals(state_signal, input_response)


class NeuronStateMachine:
    """神经元状态机"""
    
    STATES = {
        'idle': {'transitions': ['hungry', 'seeking', 'connected']},
        'hungry': {'transitions': ['eating', 'seeking']},
        'seeking': {'transitions': ['connecting', 'idle']},
        'connecting': {'transitions': ['connected', 'seeking']},
        'connected': {'transitions': ['communicating', 'idle']},
        'communicating': {'transitions': ['connected', 'idle']},
    }
    
    def evaluate(self, context):
        state = self.current_state
        score = 0
        
        if state == 'hungry':
            score += 0.3 if context.nearby_energy else 0
        elif state == 'seeking':
            score += 0.2 if context.nearby_neurons else 0
        elif state == 'connected':
            score += 0.4 if context.synapse_slots_available else 0
        
        return score
```

#### 建议的保存策略

```python
class StatePersistence:
    """状态持久化策略"""
    
    def __init__(self):
        self.snapshot_interval = 1000  # 每1000步全量快照
        self.delta_interval = 10       # 每10步增量保存
    
    def on_step(self, step_number, neurons):
        if step_number % self.snapshot_interval == 0:
            self.save_full_snapshot(neurons)
        elif step_number % self.delta_interval == 0:
            self.save_delta(neurons)
        else:
            pass  # 不保存，依赖内存状态
    
    def save_full_snapshot(self, neurons):
        """全量快照"""
        snapshot = {
            'timestamp': time.now(),
            'neurons': [n.to_dict() for n in neurons],
            'connections': self.get_all_connections(),
            'metadata': self.get_system_metadata()
        }
        self.storage.save_snapshot(snapshot)
    
    def save_delta(self, neurons):
        """增量保存"""
        changed = [n for n in neurons if n.has_changed()]
        if changed:
            self.storage.save_delta(changed)
```

#### 建议的引力模型

```python
class GravitySystem:
    """引力系统"""
    
    def __init__(self, config):
        self.R = config.sphere_radius
        self.G_center = config.center_gravity  # 中心引力常数
        self.G_mutual = config.mutual_gravity  # 神经元间引力常数
        self.damping = config.damping_factor   # 阻尼系数
    
    def calculate_total_force(self, neuron, all_neurons):
        """计算总引力"""
        total_force = Vector3(0, 0, 0)
        
        # 1. 中心引力（如果在中球外）
        distance_from_center = neuron.position.magnitude()
        if distance_from_center > self.R / 2:
            center_force = self._center_gravity(neuron, distance_from_center)
            total_force += center_force
        
        # 2. 神经元间引力（可选）
        if self.G_mutual > 0:
            for other in all_neurons:
                if other.id != neuron.id:
                    mutual_force = self._mutual_gravity(neuron, other)
                    total_force += mutual_force
        
        # 3. 阻尼（防止速度无限增加）
        total_force -= self.damping * neuron.velocity
        
        return total_force
    
    def _center_gravity(self, neuron, distance):
        """中心引力计算"""
        # 牛顿引力公式，但到R/2处停止
        direction = -neuron.position.normalized()
        magnitude = self.G_center * neuron.mass / (distance ** 2)
        return direction * magnitude
    
    def _mutual_gravity(self, neuron_a, neuron_b):
        """神经元间引力"""
        delta = neuron_b.position - neuron_a.position
        distance = delta.magnitude()
        
        if distance < MIN_DISTANCE:
            return Vector3(0, 0, 0)  # 防止除零
        
        direction = delta.normalized()
        magnitude = self.G_mutual * neuron_a.mass * neuron_b.mass / (distance ** 2)
        return direction * magnitude
```

```python
class SynapseGap:
    """突触间隙模型"""
    
    MIN_GAP = 0      # 纳米
    STANDARD_GAP = 20
    MAX_GAP = 120
    
    def __init__(self, pre_neuron, post_neuron):
        self.pre = pre_neuron
        self.post = post_neuron
        self.gap = self.STANDARD_GAP
    
    def calculate_transmission_speed(self):
        """计算信息传递速度"""
        # 间隙越小，传递越快
        # 使用指数衰减模型
        base_speed = 1.0
        decay_factor = 0.01
        
        speed = base_speed * math.exp(-decay_factor * self.gap)
        return max(0.1, min(1.0, speed))
    
    def calculate_transmission_delay(self):
        """计算传递延迟（时间步）"""
        speed = self.calculate_transmission_speed()
        return int(1 / speed)
    
    def adjust_gap(self, neuron_distance):
        """根据神经元距离调整间隙"""
        # 突触可以弹性伸缩
        # 但不能超过最大值
        ideal_gap = self.STANDARD_GAP
        actual_gap = min(self.MAX_GAP, max(self.MIN_GAP, ideal_gap))
        
        # 如果神经元距离变化，间隙也需要调整
        self.gap = actual_gap
```

## 四、逻辑矛盾

### 🔴 分裂规则的矛盾

#### 矛盾1：消亡 vs 死亡

```
文档表述：
├── "神经元永远不会因饥饿死亡"
├── 但分裂后："神经元A (消亡)"
└── 死亡？
注：└──当分裂到99次，就会保留神经元A，活着不会死亡。但是分裂次数从0开始计算。神经元B也是从0开始计算活着。分裂99次也同样会再产生一个B。
```

**分析**：这是术语定义问题。"消亡"在分裂语境下应该理解为"转化"，即原神经元转化为两个新神经元，而不是"死亡"。

#### 矛盾2：分裂条件优先级

```
条件A："饱腹度低 → 分裂倾向增加"
条件B："吃饱后33个时间步未连接 → 分裂"

注：问题：两个条件如何协调？优先级？随机就好。让神经元自己随机。让它发的信号，自己处理。如果是单个神经元，它给程序信号是啥，就给判断，无法判断，就随机一个。
```

#### 建议的分裂规则

```python
class DivisionRule:
    """分裂规则"""
    
    # 分裂类型
    HUNGER_DIVISION = 'hunger'      # 饥饿分裂
    LONELY_DIVISION = 'lonely'      # 孤独分裂
    
    def check_division(self, neuron):
        """检查是否应该分裂"""
        
        # 优先级1：饥饿分裂（生存压力）
        if neuron.satiety < 0.2:
            if neuron.steps_hungry > 100:
                return self.HUNGER_DIVISION
        
        # 优先级2：孤独分裂（繁殖需求）
        if neuron.satiety >= 0.5:
            if neuron.steps_without_connection > 66:
                return self.LONELY_DIVISION
        else:
            if neuron.steps_without_connection > 33:
                return self.LONELY_DIVISION
        
        return None
    
    def execute_division(self, neuron, division_type):
        """执行分裂"""
        # 创建两个子神经元
        child1 = neuron.clone(satiety_factor=0.5)
        child2 = neuron.clone(satiety_factor=0.5)
        
        # 位置偏移
        offset = random_unit_vector() * neuron.radius * 2
        child1.position = neuron.position + offset
        child2.position = neuron.position - offset
        
        # 原神经元"消亡"（从系统中移除）
        # 但不是"死亡"，而是"转化"
        
        return [child1, child2]
```

```python
class SignalEncoder:
    """信号编码器"""
    
    def __init__(self, encoding_type='distributed'):
        self.encoding_type = encoding_type
        self.char_to_pattern = self._build_pattern_map()
    
    def encode_chinese(self, char: str) -> List[Signal]:
        """将汉字编码为神经元信号"""
        
        if self.encoding_type == 'binary':
            return self._binary_encoding(char)
        elif self.encoding_type == 'frequency':
            return self._frequency_encoding(char)
        elif self.encoding_type == 'distributed':
            return self._distributed_encoding(char)
        else:
            raise ValueError(f"Unknown encoding: {self.encoding_type}")
    
    def _binary_encoding(self, char):
        """二进制编码"""
        codepoint = ord(char)
        binary = format(codepoint, '016b')  # 16位二进制
        return [Signal(value=int(b)) for b in binary]
    
    def _frequency_encoding(self, char):
        """频率编码"""
        codepoint = ord(char)
        frequency = codepoint / 65536.0  # 归一化到 [0, 1]
        return [Signal(frequency=frequency, duration=1)]
    
    def _distributed_encoding(self, char):
        """分布式编码（推荐）"""
        # 使用预训练的字向量或哈希映射
        pattern = self.char_to_pattern.get(char)
        if pattern is None:
            # 使用哈希生成伪随机模式
            pattern = self._hash_to_pattern(char)
        
        return [Signal(
            target_neuron_id=p['neuron_id'],
            intensity=p['intensity'],
            duration=p['duration']
        ) for p in pattern]
    
    def _hash_to_pattern(self, char):
        """使用哈希生成分布式模式"""
        import hashlib
        
        hash_val = hashlib.md5(char.encode()).hexdigest()
        pattern = []
        
        # 从哈希值生成多个目标神经元
        for i in range(0, 32, 4):
            neuron_id = int(hash_val[i:i+4], 16) % NEURON_COUNT
            intensity = int(hash_val[i+4:i+8], 16) / 65536.0
            duration = 1 + int(hash_val[i+8:i+12], 16) % 5
            
            pattern.append({
                'neuron_id': neuron_id,
                'intensity': intensity,
                'duration': duration
            })
        
        return pattern


class SignalDecoder:
    """信号解码器"""
    
    def __init__(self, encoder):
        self.encoder = encoder
        self.pattern_to_char = self._build_reverse_map()
    
    def decode_output(self, signals: List[Signal]) -> str:
        """将神经元输出解码为字符"""
        
        if self.encoder.encoding_type == 'binary':
            return self._binary_decoding(signals)
        elif self.encoder.encoding_type == 'frequency':
            return self._frequency_decoding(signals)
        elif self.encoder.encoding_type == 'distributed':
            return self._distributed_decoding(signals)
    
    def _distributed_decoding(self, signals):
        """分布式解码"""
        # 提取模式特征
        pattern = self._extract_pattern(signals)
        
        # 查找最匹配的字符
        best_match = None
        best_score = 0
        
        for char, char_pattern in self.pattern_to_char.items():
            score = self._pattern_similarity(pattern, char_pattern)
            if score > best_score:
                best_score = score
                best_match = char
        
        if best_score > MATCH_THRESHOLD:
            return best_match
        else:
            return None  # 无法识别
```

```python
class MeaningfulOutputDetector:
    """有意义输出检测器"""
    
    def __init__(self):
        self.chinese_dict = self._load_chinese_dict()
        self.pattern_history = []
    
    def detect(self, output: str, context: dict) -> dict:
        """检测输出是否有意义"""
        
        results = {
            'is_meaningful': False,
            'type': None,
            'confidence': 0.0,
            'details': {}
        }
        
        # 检测1：中文词语
        chinese_result = self._detect_chinese_words(output)
        if chinese_result['found']:
            results['is_meaningful'] = True
            results['type'] = 'chinese_words'
            results['confidence'] = chinese_result['confidence']
            results['details'] = chinese_result
            return results
        
        # 检测2：数字
        number_result = self._detect_numbers(output)
        if number_result['found']:
            results['is_meaningful'] = True
            results['type'] = 'numbers'
            results['confidence'] = number_result['confidence']
            results['details'] = number_result
            return results
        
        # 检测3：重复模式
        pattern_result = self._detect_repeated_pattern(output)
        if pattern_result['found']:
            results['is_meaningful'] = True
            results['type'] = 'repeated_pattern'
            results['confidence'] = pattern_result['confidence']
            results['details'] = pattern_result
            return results
        
        # 检测4：与输入相关
        if 'input' in context:
            relevance_result = self._detect_input_relevance(output, context['input'])
            if relevance_result['found']:
                results['is_meaningful'] = True
                results['type'] = 'input_relevant'
                results['confidence'] = relevance_result['confidence']
                results['details'] = relevance_result
                return results
        
        return results
    
    def _detect_chinese_words(self, output):
        """检测中文词语"""
        import re
        
        # 提取所有中文字符
        chinese_chars = re.findall(r'[\u4e00-\u9fff]+', output)
        
        if not chinese_chars:
            return {'found': False}
        
        # 检查是否在词典中
        found_words = []
        for chars in chinese_chars:
            if chars in self.chinese_dict:
                found_words.append(chars)
        
        if found_words:
            return {
                'found': True,
                'words': found_words,
                'confidence': len(found_words) / len(chinese_chars)
            }
        
        return {'found': False}
    
    def _detect_repeated_pattern(self, output):
        """检测重复模式"""
        # 使用自相关检测重复模式
        min_pattern_len = 2
        max_pattern_len = len(output) // 2
        
        for pattern_len in range(min_pattern_len, max_pattern_len + 1):
            pattern = output[:pattern_len]
            repetitions = output.count(pattern)
            
            if repetitions >= 3:  # 重复3次以上
                return {
                    'found': True,
                    'pattern': pattern,
                    'repetitions': repetitions,
                    'confidence': min(1.0, repetitions / 5)
                }
        
        return {'found': False}
    
    def _detect_input_relevance(self, output, input_text):
        """检测与输入的相关性"""
        # 简单的相关性检测
        # 可以使用编辑距离、共同子串等方法
        
        common_chars = set(output) & set(input_text)
        
        if len(common_chars) > 0:
            return {
                'found': True,
                'common_chars': list(common_chars),
                'confidence': len(common_chars) / max(len(output), 1)
            }
        
        return {'found': False}
```
